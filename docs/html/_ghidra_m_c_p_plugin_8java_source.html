752 private String renameVariableInFunction(String functionName, String oldVarName, String newVarName) {
753 Program program = getCurrentProgram();
754 if (program == null) return "No program loaded";
755
756 DecompInterface decomp = new DecompInterface();
757 decomp.openProgram(program);
758
759 Function func = null;
760 for (Function f : program.getFunctionManager().getFunctions(true)) {
761 if (f.getName().equals(functionName)) {
762 func = f;
763 break;
764 }
765 }
766
767 if (func == null) {
768 return "Function not found";
769 }
770
771 DecompileResults result = decomp.decompileFunction(func, 30, new ConsoleTaskMonitor());
772 if (result == null || !result.decompileCompleted()) {
773 return "Decompilation failed";
774 }
775
776 HighFunction highFunction = result.getHighFunction();
777 if (highFunction == null) {
778 return "Decompilation failed (no high function)";
779 }
780
781 LocalSymbolMap localSymbolMap = highFunction.getLocalSymbolMap();
782 if (localSymbolMap == null) {
783 return "Decompilation failed (no local symbol map)";
784 }
785
786 HighSymbol highSymbol = null;
787 Iterator<HighSymbol> symbols = localSymbolMap.getSymbols();
788 while (symbols.hasNext()) {
789 HighSymbol symbol = symbols.next();
790 String symbolName = symbol.getName();
791
792 if (symbolName.equals(oldVarName)) {
793 highSymbol = symbol;
794 }
795 if (symbolName.equals(newVarName)) {
796 return "Error: A variable with name '" + newVarName + "' already exists in this function";
797 }
798 }
799
800 if (highSymbol == null) {
801 return "Variable not found";
802 }
803
804 boolean commitRequired = checkFullCommit(highSymbol, highFunction);
805
806 final HighSymbol finalHighSymbol = highSymbol;
807 final Function finalFunction = func;
808 AtomicBoolean successFlag = new AtomicBoolean(false);
809
810 try {
811 SwingUtilities.invokeAndWait(() -> {
812 int tx = program.startTransaction("Rename variable");
813 try {
814 if (commitRequired) {
815 HighFunctionDBUtil.commitParamsToDatabase(highFunction, false,
816 ReturnCommitOption.NO_COMMIT, finalFunction.getSignatureSource());
817 }
818 HighFunctionDBUtil.updateDBVariable(
819 finalHighSymbol,
820 newVarName,
821 null,
822 SourceType.USER_DEFINED
823 );
824 successFlag.set(true);
825 }
826 catch (Exception e) {
827 Msg.error(this, "Failed to rename variable", e);
828 }
829 finally {
830 successFlag.set(program.endTransaction(tx, true));
831 }
832 });
833 } catch (InterruptedException | InvocationTargetException e) {
834 String errorMsg = "Failed to execute rename on Swing thread: " + e.getMessage();
835 Msg.error(this, errorMsg, e);
836 return errorMsg;
837 }
838 return successFlag.get() ? "Variable renamed" : "Failed to rename variable";

850 protected static boolean checkFullCommit(HighSymbol highSymbol, HighFunction hfunction) {
851 if (highSymbol != null && !highSymbol.isParameter()) {
852 return false;
853 }
854 Function function = hfunction.getFunction();
855 Parameter[] parameters = function.getParameters();
856 LocalSymbolMap localSymbolMap = hfunction.getLocalSymbolMap();
857 int numParams = localSymbolMap.getNumParams();
858 if (numParams != parameters.length) {
859 return true;
860 }
861
862 for (int i = 0; i < numParams; i++) {
863 HighSymbol param = localSymbolMap.getParamSymbol(i);
864 if (param.getCategoryIndex() != i) {
865 return true;
866 }
867 VariableStorage storage = param.getStorage();
868 // Don't compare using the equals method so that DynamicVariableStorage can match

1623 private DataType resolveDataType(DataTypeManager dtm, String typeName) {
1624 // First try to find exact match in all categories
1625 DataType dataType = findDataTypeByNameInAllCategories(dtm, typeName);
1626 if (dataType != null) {
1627 Msg.info(this, "Found exact data type match: " + dataType.getPathName());
1628 return dataType;
1629 }
1630
1631 // Check for Windows-style pointer types (PXXX)
1632 if (typeName.startsWith("P") && typeName.length() > 1) {
1633 String baseTypeName = typeName.substring(1);
1634
1635 // Special case for PVOID
1636 if (baseTypeName.equals("VOID")) {
1637 return new PointerDataType(dtm.getDataType("/void"));
1638 }
1639
1640 // Try to find the base type
1641 DataType baseType = findDataTypeByNameInAllCategories(dtm, baseTypeName);
1642 if (baseType != null) {
1643 return new PointerDataType(baseType);
1644 }
1645
1646 Msg.warn(this, "Base type not found for " + typeName + ", defaulting to void*");
1647 return new PointerDataType(dtm.getDataType("/void"));
1648 }
1649
1650 // Handle common built-in types
1651 switch (typeName.toLowerCase()) {
1652 case "int":
1653 case "long":
1654 return dtm.getDataType("/int");
1655 case "uint":
1656 case "unsigned int":
1657 case "unsigned long":
1658 case "dword":
1659 return dtm.getDataType("/uint");
1660 case "short":
1661 return dtm.getDataType("/short");
1662 case "ushort":
1663 case "unsigned short":
1664 case "word":
1665 return dtm.getDataType("/ushort");
1666 case "char":
1667 case "byte":
1668 return dtm.getDataType("/char");
1669 case "uchar":
1670 case "unsigned char":
1671 return dtm.getDataType("/uchar");
1672 case "longlong":
1673 case "__int64":
1674 return dtm.getDataType("/longlong");
1675 case "ulonglong":
1676 case "unsigned __int64":
1677 return dtm.getDataType("/ulonglong");
1678 case "bool":
1679 case "boolean":
1680 return dtm.getDataType("/bool");
1681 case "void":
1682 return dtm.getDataType("/void");
1683 default:
1684 // Try as a direct path
1685 DataType directType = dtm.getDataType("/" + typeName);
1686 if (directType != null) {
1687 return directType;
1688 }
